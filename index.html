// ============================================================ // SCADENZIARIO SPINEL CAFFÈ — Google Apps Script Backend v4 // Supporto JSONP per bypassare CORS senza proxy // ============================================================ const SHEET_NAME = 'Scadenze'; const HEADERS = [ 'id', 'titolo', 'ente', 'tipo', 'scadenza', 'importo', 'note', 'reminders', 'completato', 'visibility', 'owner', 'updatedAt' ]; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { const params = e.parameter || {}; let body = {}; if (e.postData) { try { body = JSON.parse(e.postData.contents || '{}'); } catch(ex) {} } const action = params.action || body.action; const owner = params.owner || body.owner || null; const callback = params.callback || null; // JSONP callback let result; if (action === 'getAll') { result = getAllItems(owner); } else if (action === 'save') { let item = body.item; if (!item && params.item) { try { item = JSON.parse(decodeURIComponent(params.item)); } catch(ex) {} } result = saveItem(item); } else if (action === 'delete') { const id = params.id || body.id; const delOwner = params.owner || body.owner; result = deleteItem(id, delOwner); } else { result = { error: 'Azione non riconosciuta: ' + action }; } const json = JSON.stringify(result); // Se richiesta JSONP, wrappa nel callback if (callback) { return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } catch (err) { const errJson = JSON.stringify({ error: err.message }); const callback = (e.parameter || {}).callback; if (callback) { return ContentService .createTextOutput(callback + '(' + errJson + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(errJson) .setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); sheet.getRange(1, 1, 1, HEADERS.length).setFontWeight('bold'); sheet.setFrozenRows(1); } return sheet; } function rowToObj(headers, row) { const obj = {}; headers.forEach((h, i) => { if (h === 'reminders') { try { obj[h] = JSON.parse(row[i] || '[]'); } catch(e) { obj[h] = []; } } else if (h === 'completato') { obj[h] = row[i] === true || row[i] === 'true'; } else { obj[h] = row[i]; } }); return obj; } function getAllItems(owner) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { items: [] }; const headers = data[0]; const items = data.slice(1) .map(row => rowToObj(headers, row)) .filter(item => { if (!item.id) return false; if (item.visibility === 'shared' || !item.visibility) return true; if (item.visibility === 'personal') return owner && String(item.owner) === String(owner); return false; }); return { items }; } function saveItem(item) { if (!item || !item.id) return { error: 'Item non valido' }; const sheet = getSheet(); const data = sheet.getDataRange().getValues(); item.updatedAt = new Date().toISOString(); const row = HEADERS.map(h => { if (h === 'reminders') return JSON.stringify(item[h] || []); return item[h] !== undefined ? item[h] : ''; }); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(item.id)) { sheet.getRange(i + 1, 1, 1, HEADERS.length).setValues([row]); return { ok: true, action: 'updated' }; } } sheet.appendRow(row); return { ok: true, action: 'inserted' }; } function deleteItem(id, owner) { if (!id) return { error: 'ID mancante' }; const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const headers = data[0]; for (let i = data.length - 1; i >= 1; i--) { const item = rowToObj(headers, data[i]); if (String(item.id) === String(id)) { if (item.visibility === 'personal' && String(item.owner) !== String(owner)) { return { error: 'Non autorizzato' }; } sheet.deleteRow(i + 1); return { ok: true }; } } return { error: 'Item non trovato' }; }
